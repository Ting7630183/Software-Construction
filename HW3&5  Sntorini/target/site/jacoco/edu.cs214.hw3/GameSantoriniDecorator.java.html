package edu.cs214.hw3;

import java.util.List;

public class GameSantoriniDecorator implements GameSantorini{
    GameSantoriniComponent component;
    Board board;
    Player player;
    Player nextPlayer;
    boolean lastUp;
    int[] lastBuildDirection;
    String godCard;

    public GameSantoriniDecorator(GameSantoriniComponent component, Board board, Player player, Player nextPlayer, boolean lastUp, int[] lastBuildDirection, String godCard){
        this.component = component;
        this.board = board;
        this.player = player;
        this.nextPlayer = nextPlayer;
        this.lastUp = lastUp;
        this.lastBuildDirection = lastBuildDirection;
        this.godCard = godCard;
    }

    @Override
    public boolean checkWinner() {
        return component.checkWinner();
    }

    @Override
    public List<List<Integer>> checkNextMovablePosition(Player player) {
        return component.checkNextMovablePosition(player);
    }

    @Override
    public boolean checkMoveValidity(int x, int y, int[] direction) {
        return component.checkMoveValidity(x, y, direction);
    }

    @Override
    public boolean checkBuildValidity(int x, int y, int[] direction) {
        return component.checkBuildValidity(x, y, direction);
    }

    @Override
    public void move(int x, int y, int[] direction, Player nextPlayer) {
        component.move(x, y, direction, nextPlayer);
    }

    @Override
    public Board build(int x, int y, int[] direction, boolean block) {
        return component.build(x, y, direction, block);
    }

    public void updateLastUp(int x, int y, int[] direction){
        int xx = x + direction[0];
        int yy = y + direction[1];
        int currentHeight = this.board.getCell(x, y);
        int nextHeight = this.board.getCell(xx, yy);
        if((nextHeight - currentHeight) == 1){
            this.lastUp = true;
        }else{
            this.lastUp = false;
        }
    }

    @Override
    public void putWorker(int x, int y, Player player) {
        component.putWorker(x, y, player);
    }

    public boolean getLastUp(){
        return this.lastUp;
    }

    public int[] getLastBuildDirection(){
        return this.lastBuildDirection;
    }
}
