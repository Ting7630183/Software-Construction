package edu.cs214.hw3;

public class GameWithDemeter extends GameSantoriniDecorator{
    String godCard = "Demeter";
    int[] lastBuildDirection;
    public GameWithDemeter(GameSantoriniComponent component, Board board, Player player, Player nextPlayer, boolean lastUp, int[] lastBuildDirection) {
        super(component, board, player, nextPlayer, lastUp, lastBuildDirection, "demeter");
    }

    @Override
    public boolean checkBuildValidity(int x, int y, int[] direction) {
        return super.checkBuildValidity(x, y, direction) && checkLastBuildSame(direction, this.lastBuildDirection);
    }

    private boolean checkLastBuildSame(int[] direction, int[] lastBuildDirection){
        if(direction[0] == lastBuildDirection[0] && direction[1] == lastBuildDirection[1]){
            return false;
        }
        return true;
    }
}
