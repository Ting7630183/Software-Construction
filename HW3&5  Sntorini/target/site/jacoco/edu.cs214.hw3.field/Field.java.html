package edu.cs214.hw3.field;

public class Field {
    int height;
    String workerName;

    public Field(int height, String workerName){
        this.height = height;
        this.workerName = workerName;
    }
    /** get the height of the field*/
    public int getHeight(){
        return this.height;
    }
    /**set workerName*/
    public void setWorkerName(String name){
        this.workerName = name;
    }

    /**get the worker's name of this field*/
    public String getWorkerName(){
        return this.workerName;
    }

    /**add block to this field change the height*/
    public void addBlock(){
        if(this.height < 3){
            this.height += 1;
        }else{
            return;
        }
    }

    /**add doom to this field and change the height*/
    public boolean addDoom(){
        if(this.height == 3){
            this.height += 1;
            return true;
        }
        return false;
    }

    /**if there is worker comes to this field, update the field's workerName as the worker's name, if the worker leave this filed, the workerName become null*/
    public void changeWorkerName(String newWorkerName){
        this.workerName = newWorkerName;
    }
}
