package edu.cs214.hw3;

import java.util.ArrayList;
import java.util.List;

public class GameWithMinotaur extends GameSantoriniDecorator{
    String godCard = "Minotaur";


    public GameWithMinotaur(GameSantoriniComponent component, Board board, Player player, Player nextPlayer, Player player1, boolean lastUp, int[] lastBuildDirection) {
        super(component, board, player, nextPlayer, lastUp, lastBuildDirection, "Minotaur");
    }

    @Override
    public boolean checkMoveValidity(int x, int y, int[] direction) {
        return checkOccupancy(x, y, direction) &&
                checkHeightDifference(x, y, direction) &&
                checkAdjacency(x, y, direction) &&
                checkWithinBoard(x, y, direction);
    }

    private boolean checkOccupancy(int x, int y, int[] direction){
        int xx = x + direction[0];
        int yy = y + direction[1];
        if(this.board.getPlayer(xx, yy) == null){
            return true;
        }else{
            if(!this.board.getPlayer(xx, yy ).equals(this.player)){
                int xxx = x + direction[0] + direction[0];
                int yyy = y + direction[1] + direction[1];
                if(this.board.getPlayer(xxx, yyy) == null){
                    return true;
                }
            }
        }
        return false;
    }

    private boolean checkHeightDifference(int x, int y, int[] direction){
        int heightDifference = getHeightDifference(x, y, direction);
        if(heightDifference <= 1){
            return true;
        }
        return false;
    }

    private int getHeightDifference(int x, int y, int[] direction){
        int currentHeight = this.board.getCell(x, y);
        int nextHeight = this.board.getCell(direction[0], direction[1]);
        return nextHeight - currentHeight;
    }


    private boolean checkWithinBoard(int x, int y, int[] direction){
        int xx = x + direction[0];
        int yy = y + direction[1];
        if(xx < 5 && yy < 5){
            return true;
        }
        return false;
    }

    private boolean checkAdjacency(int x, int y, int[] direction){
        for(int i = 0; i < 2; i++){
            if(Math.abs(direction[i]) > 1){
                return false;
            }
        }
        return true;
    }

    @Override
    public void move(int x, int y, int[] direction, Player nextPlayer) {
        int xx = x + direction[0];
        int yy = y + direction[1];
        super.move(xx, yy, direction, this.nextPlayer);
        super.move(x, y, direction, this.player);

    }
}
