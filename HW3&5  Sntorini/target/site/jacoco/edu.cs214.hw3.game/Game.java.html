package edu.cs214.hw3.game;

import edu.cs214.hw3.field.Field;
import edu.cs214.hw3.grid.Grid;
import edu.cs214.hw3.players.Player;


public class Game {
    Player[] players;
    Grid grid;

    public Game(Player[] players, Grid grid){
        this.players = players;
        this.grid = grid;
    }
    public void newGame(){
        Grid grid = new Grid();

    }

    public void setWorkerInitialPosition(int[][] initialPosition){
        /** check whether all initial position are within the grid boundary of 5*5*/
        for(int i = 0; i < initialPosition.length; i++){
            for(int j = 0; j < initialPosition[i].length; j++){
                if(initialPosition[i][j] < 0 || initialPosition[i][j] > 5){
                    return;
                }
            }
        }
        /**check whether the position has been occupied, if not put the worker to the position*/
        String[] workerName = new String[]{"player1Worker1", "player1Worker2", "playerWorker1", "playerWorker2"};
        for(int i = 0; i < initialPosition.length; i++){
            Field field = grid.getField(initialPosition[i][0], initialPosition[i][1]);
            if(field.getWorkerName() != null){
                return;
            }else {
                field.setWorkerName(workerName[i]);
            }
        }
    }

    public void move(int[]workerPosition, int[] direction){
        /** check the validity of the move, including direction, high difference, within the boundary or not, whether the original field has worker or not*/
        boolean validity = grid.checkMoveValidity(workerPosition, direction);

        /**if the move is valid, update the grid status, if not valid, return*/
        if(validity) {
            grid.updateGridStatus(workerPosition, direction);
        }else{
            return;
        }
    }

    public void build(int[] workerPosition, int[] direction, boolean dome, boolean block) {
        /**Check whether is to move the worker to adjacent field within the grid*/
        boolean buildValidity = grid.checkBuildValidity(workerPosition, direction);

        if (!buildValidity) {
            grid.buildTower(workerPosition, direction, dome, block);
        } else{
            return;
        }
    }
}
