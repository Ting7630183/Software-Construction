package edu.cs214.hw3;

public class GameWithPan extends GameSantoriniDecorator{
    int x;
    int y;
    int[] direction;
    String godCard = "Pan";

    public GameWithPan(GameSantoriniComponent component, Board board, Player player, Player nextPlayer, boolean lastUp, int[] lastBuildDirection) {
        super(component, board, player, nextPlayer, lastUp, lastBuildDirection, "pan");
    }

    @Override
    public boolean checkWinner() {
        if(MoveDownTwoOrMore(this.x, this.y, this.direction)){
            return true;
        }else {
            return super.checkWinner();
        }
    }

    private boolean MoveDownTwoOrMore(int x, int y, int[] direction){
        int currentHeight = this.board.getCell(x, y);
        int nextHeight = this.board.getCell(x+direction[0], y+direction[1]);
        if(currentHeight - nextHeight >=2){
            return true;
        }
        return false;
    }

}
