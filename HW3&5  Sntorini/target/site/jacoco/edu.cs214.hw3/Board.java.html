package edu.cs214.hw3;

import java.util.ArrayList;
import java.util.List;

public class Board {
    int[][] cell;
    Player[][] players;

    public Board(){
        cell = new int[5][5];
        for(int i = 0; i < 5; i++){
            for(int j = 0; j < 5; j++){
                cell[i][j] = 0;
            }
        }
        players = new Player[5][5];
        for(int i = 0; i < 5; i++){
            for(int j = 0; j < 5; j++){
                players[i][j] = null;
            }
        }
    }

    public Board(int[][] cell, Player[][] players){
        this.cell = cell;
        this.players = players;
    }

    public int getCell(int x, int y){
        return cell[x][y];
    }

    public void updateCell(int x, int y, boolean block) {
        if (block == true) {
            cell[x][y] = getCell(x, y) + 1;
        }else{
            cell[x][y] = 4;
        }
    }

    public Player getPlayer(int x, int y){
        return players[x][y];
    }

    public void updatePlayer(int x, int y, Player player){
        this.players[x][y] = player;
    }

    public List<List<Integer>> getPlayerPosition(Player player){
        if(player == null){
            return new ArrayList<>();
        }
        List<List<Integer>> playerPosition = new ArrayList<>();
        for(int i = 0; i < 5; i++){
            for(int j = 0; j < 5; j++){
                if(this.getPlayer(i, j) != null && this.getPlayer(i, j).equals(player)){
                    List<Integer> list = new ArrayList<Integer>();
                    list.add(i);
                    list.add(j);
                    playerPosition.add(list);
                }
            }
        }
        return playerPosition;
    }
}
